public async Task<OsintResult> ExecuteAsync(string target, string investigationType, CancellationToken cancellationToken)     {         var apiKey = _configuration["SpiderFoot:ApiKey"];         var baseUrl = _configuration["SpiderFoot:BaseUrl"];


// Start scan         var startResponse = await _httpClient.PostAsync(             $"{baseUrl}/scan/new?apikey={apiKey}&scanTarget={target}&scanType=domain",             null, cancellationToken);


if (!startResponse.IsSuccessStatusCode)             throw new Exception($"SpiderFoot API error: {startResponse.StatusCode}");


var scanId = await startResponse.Content.ReadAsStringAsync();


// Poll for results (implementation simplified for brevity)         return new OsintResult         {             ToolName = ToolName,             DataType = "DomainIntelligence",             RawData = await GetScanResults(scanId, cancellationToken),             Summary = $"Domain intelligence for {target}",             CollectedAt = DateTime.UtcNow         };     }


public bool SupportsOperation(string operationType)          => operationType == "DomainScan"; }

4.2 Integration Service

public class OsintIntegrationService : IOsintIntegrationService {     private readonly IEnumerable<IOsintPlugin> _plugins;     private readonly OsintDbContext _context;     private readonly ILogger<OsintIntegrationService> _logger;


public OsintIntegrationService(         IEnumerable<IOsintPlugin> plugins,         OsintDbContext context,         ILogger<OsintIntegrationService> logger)     {         _plugins = plugins;         _context = context;         _logger = logger;     }


public async Task ExecuteDomainScan(int investigationId, string target)     {         try         {             var investigation = await _context.Investigations.FindAsync(investigationId);             if (investigation == null) return;


investigation.Status = InvestigationStatus.Running;             await _context.SaveChangesAsync();


var domainPlugins = _plugins.Where(p => p.SupportsOperation("DomainScan"));


foreach (var plugin in domainPlugins)             {                 if (!plugin.IsEnabled) continue;


try                 {                     var result = await plugin.ExecuteAsync(target, "DomainScan", CancellationToken.None);                     result.OsintInvestigationId = investigationId;


_context.Results.Add(result);                     await _context.SaveChangesAsync();                 }                 catch (Exception ex)                 {                     _logger.LogError(ex, "Error executing plugin {Plugin} for target {Target}", plugin.ToolName, target);                 }             }


investigation.Status = InvestigationStatus.Completed;             await _context.SaveChangesAsync();         }         catch (Exception ex)         {             _logger.LogError(ex, "Error executing domain scan for investigation {InvestigationId}", investigationId);         }     } }

Ollama vs LM Studio: Technical Rationale

Here's my analysis of why Ollama is the better choice for this project:

Ollama Advantages for OSINT Framework:

  1. Docker-Native Design

- Official Docker image available - Built for headless server environments - Easy container orchestration - Lightweight (~150MB base image)

  1. API-First Architecture

   # Simple REST API calls    curl -X POST http://ollama:11434/api/generate -d '{      "model": "llama2",      "prompt": "Analyze this OSINT data...",      "stream": false    }'

  1. Model Management

- Built-in model repository - Easy model downloading/pulling - Version management - Multiple model support

  1. Linux Optimization

- Native Linux performance - GPU acceleration support - Minimal resource overhead - Production-ready

LM Studio Limitations:

  1. Desktop-Focused

- Designed as GUI application - No official Docker support - Hard to automate in CI/CD

  1. Windows/Mac Focused

- Limited Linux support - Not optimized for servers - Resource intensive

  1. API Limitations

- Secondary feature to GUI - Less stable in headless mode - Limited configuration options

Alternative Considerations:

text-generation-webui

  • Pros: Feature-rich, many models
  • Cons: Heavy, complex API

LocalAI

  • Pros: OpenAI API compatible
  • Cons: More complex setup

  1. Scalability: Run multiple Ollama instances

  3. Updates: Easy version updates via Docker
  4. Resource Management: Control CPU/GPU allocation

For an OSINT framework that needs to run reliably in Docker on Linux with easy AI integration, Ollama is objectively the better technical choice. LM Studio is excellent for desktop experimentation but isn't designed for production server deployment.

## Part 2: Remote AI Instance (Ollama) Setup

Yes, it is essentially a matter of IP address and port for the Ollama API. However, there are a few considerations:

### Steps:

1. **Install Ollama on the remote machine** (which could be another Debian server or any supported OS) by following the instructions from [Ollama website](https://ollama.ai/).

For Linux, you can use:
```bash
curl -fsSL https://ollama.com/install.sh | sh
```

2. **Start the Ollama service** and ensure it's running on the default port (11434).

By default, Ollama runs on `127.0.0.1:11434`. To allow remote connections, you need to change the service to listen on `0.0.0.0`.

You can set the environment variable `OLLAMA_HOST` to `0.0.0.0` before starting the service.

Edit the systemd service file for Ollama (if installed via the script) or create an override.

Alternatively, you can set the environment variable in the service file.

Example for systemd:

Create a drop-in directory for the ollama service:
```bash
sudo mkdir -p /etc/systemd/system/ollama.service.d
```

Create a file named `override.conf` in that directory with:
```ini
[Service]
Environment="OLLAMA_HOST=0.0.0.0"
```

Then reload systemd and restart Ollama:
```bash
sudo systemctl daemon-reload
sudo systemctl restart ollama
```

3. **Firewall Configuration**: Ensure the port 11434 is open on the remote machine's firewall.

For UFW (if enabled):
```bash
sudo ufw allow 11434/tcp
```

4. **Test the remote connection** from your development machine:

```bash
curl http://<remote-ip>:11434/api/tags
```

You should get a JSON response with the available models.

5. **Update the application configuration** to use the remote Ollama instance.

In the `docker-compose.yml` file for the backend service, change the environment variable `Ollama__BaseUrl` to the remote machine's IP and port:

```yaml
environment:
- Ollama__BaseUrl=http://<remote-ip>:11434
```

Alternatively, you can set this in the `appsettings.json` or via environment variables in the container.

6. **Security Consideration**: Since Ollama does not have built-in authentication, exposing the service on a public IP without any protection is a security risk. Consider:

- Running Ollama on a private network.
- Using a reverse proxy with authentication (like nginx with basic auth) in front of Ollama.
- Using a VPN to access the remote machine.

### Example of using a remote Ollama in the backend service:

In the `docker-compose.yml` for the backend service, you would have:

```yaml
backend:
# ... other configurations ...
environment:
- Ollama__BaseUrl=http://192.168.1.100:11434 # Replace with the remote IP
```

Then, when the backend service runs, it will use the remote Ollama instance.
