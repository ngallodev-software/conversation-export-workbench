run this and create a graph jpeg or png import matplotlib.pyplot as plt import numpy as np

Data for the chart

charges = np.array([100, 250, 437, 750]) # Grains v_12in = np.array([350, 580, 850, 1000]) # FPS v_24in = np.array([420, 710, 1050, 1250]) # FPS v_36in = np.array([460, 790, 1150, 1380]) # FPS plt.figure(figsize=(10, 6))

Plotting the lines

plt.plot(charges, v_12in, marker='o', linestyle='--', color='red', label='12" Barrel (Short)') plt.plot(charges, v_24in, marker='s', linestyle='-', color='blue', linewidth=2, label='24" Barrel (Optimized)') plt.plot(charges, v_36in, marker='^', linestyle='--', color='green', label='36" Barrel (Long)')

Speed of Sound Reference

plt.axhline(y=1125, color='black', linestyle=':', alpha=0.5) plt.text(110, 1140, 'Speed of Sound (~1125 fps)', fontsize=10, color='black')

Formatting

plt.title('Estimated Muzzle Velocity: Golf Ball Cannon\n(1Fg Black Powder | 1.75" ID Aluminum)', fontsize=14) plt.xlabel('Black Powder Charge (Grains)', fontsize=12) plt.ylabel('Muzzle Velocity (Feet Per Second)', fontsize=12) plt.legend() plt.grid(True, which='both', linestyle=':', alpha=0.6) plt.tight_layout() plt.show()
