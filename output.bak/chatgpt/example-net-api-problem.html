```csharp using System; using System.Collections.Generic; using System.Net.Http; using System.Text.Json; using System.Threading.Tasks;

public class OrdersApiResponse { public int page { get; set; } public int per_page { get; set; } public int total { get; set; } public int total_pages { get; set; } public List data { get; set; } = new(); }

public class OrderDto { public int id { get; set; } public string customer { get; set; } = string.Empty; public string orderDate { get; set; } = string.Empty; public string status { get; set; } = string.Empty; public decimal amount { get; set; } }

public class Solution { private const string BaseUrl = "https://jsonmock.hackerrank.com/api/orders";

public static async Task GetTotalRevenueAsync(int year) { using var client = new HttpClient();

var totalRevenue = 0m; var page = 1; var totalPages = 1;

var options = new JsonSerializerOptions { PropertyNameCaseInsensitive = true };

while (page <= totalPages) { var url = $"{BaseUrl}?status=delivered&page={page}";

var response = await client.GetAsync(url); response.EnsureSuccessStatusCode();

var json = await response.Content.ReadAsStringAsync(); var apiResponse = JsonSerializer.Deserialize(json, options);

if (apiResponse == null) { break; // or throw; in HackerRank you can often assume this won't be null }

totalPages = apiResponse.total_pages;

foreach (var order in apiResponse.data) { if (DateTime.TryParse(order.orderDate, out var date)) { if (date.Year == year) { totalRevenue += order.amount; } } }

page++; }

return totalRevenue; } } ```

Key Points in the Code

  • PropertyNameCaseInsensitive = true lets us use either page or Page in C#.
  • EnsureSuccessStatusCode() is a quick way to fail fast if HTTP status is not 2xx.
  • Pagination is handled via while (page <= totalPages) and updating totalPages from the first (or each) response.
  • DateTime.TryParse is used instead of Parse to avoid crashing on invalid dates (safer).

5. Common Pitfalls on These Questions

  1. Ignoring pagination

Only fetching page=1 will fail test cases where data spans multiple pages.

  1. Hardcoding page count

Always use total_pages from the response instead of guessing how many pages exist.

  1. Wrong property names

- JSON: orderDate - C#: OrderDate or orderDate If you mismatch, you must either: - Use exact names, or - Enable case-insensitive matching, or - Use attributes like [JsonPropertyName("orderDate")].

  1. Using int for money

Amounts are usually decimal. Use decimal for currency to avoid precision issues.

  1. Synchronous HTTP in async environment

Avoid .Result / .Wait() in these environments – prefer async/await as above.

4. Full C# Solution (Walkthrough Code)

Below is a complete example you could adapt in an online test. In HackerRank, you would usually place this inside the given class (e.g. Solution), and sometimes they call it from Main.

```csharp using System; using System.Collections.Generic; using System.Net.Http; using System.Text.Json; using System.Threading.Tasks;

public class OrdersApiResponse { public int page { get; set; } public int per_page { get; set; } public int total { get; set; } public int total_pages { get; set; } public List data { get; set; } = new(); }

public class OrderDto { public int id { get; set; } public string customer { get; set; } = string.Empty; public string orderDate { get; set; } = string.Empty; public string status { get; set; } = string.Empty; public decimal amount { get; set; } }

public class Solution { private const string BaseUrl = "https://jsonmock.hackerrank.com/api/orders";

public static async Task GetTotalRevenueAsync(int year) { using var client = new HttpClient();

var totalRevenue = 0m; var page = 1; var totalPages = 1;

var options = new JsonSerializerOptions { PropertyNameCaseInsensitive = true };

while (page <= totalPages) { var url = $"{BaseUrl}?status=delivered&page={page}";

var response = await client.GetAsync(url); response.EnsureSuccessStatusCode();

var json = await response.Content.ReadAsStringAsync(); var apiResponse = JsonSerializer.Deserialize(json, options);

if (apiResponse == null) { break; // or throw; in HackerRank you can often assume this won't be null }

totalPages = apiResponse.total_pages;

foreach (var order in apiResponse.data) { if (DateTime.TryParse(order.orderDate, out var date)) { if (date.Year == year) { totalRevenue += order.amount; } } }

page++; }

return totalRevenue; } } ```

Key Points in the Code

  • PropertyNameCaseInsensitive = true lets us use either page or Page in C#.
  • EnsureSuccessStatusCode() is a quick way to fail fast if HTTP status is not 2xx.
  • Pagination is handled via while (page <= totalPages) and updating totalPages from the first (or each) response.
  • DateTime.TryParse is used instead of Parse to avoid crashing on invalid dates (safer).

5. Common Pitfalls on These Questions

  1. Ignoring pagination

Only fetching page=1 will fail test cases where data spans multiple pages.

  1. Hardcoding page count

Always use total_pages from the response instead of guessing how many pages exist.

  1. Wrong property names

- JSON: orderDate - C#: OrderDate or orderDate If you mismatch, you must either: - Use exact names, or - Enable case-insensitive matching, or - Use attributes like [JsonPropertyName("orderDate")].

  1. Using int for money

Amounts are usually decimal. Use decimal for currency to avoid precision issues.

  1. Synchronous HTTP in async environment

Avoid .Result / .Wait() in these environments – prefer async/await as above.
